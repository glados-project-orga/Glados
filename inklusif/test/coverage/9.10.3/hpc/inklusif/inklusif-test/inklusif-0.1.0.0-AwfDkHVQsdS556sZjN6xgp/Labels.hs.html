    8 module Labels (
    9   LabelTable,
   10   parseLabel,
   11   generateLabel,
   12   resolveLabels
   13 ) where
   14 
   15 import qualified Data.Map as M
   16 import CompilerTypes (CompilerData)
   17 
   18 type LabelTable = M.Map String Int
   19 
   20 parseLabel :: String -> Maybe String
   21 parseLabel line =
   22   case span (/= ':') line of
   23     (('L':rest), ":") -> Just ('L':rest)
   24     _                 -> Nothing
   25 
   26 isDirective :: String -> Bool
   27 isDirective line =
   28   case words line of
   29     ["header", "{"]     -> True
   30     ["}"]               -> True
   31     ["fun", _name, "{"] -> True
   32     _                   -> False
   33 
   34 correspondLine :: LabelTable -> String -> String
   35 correspondLine table line =
   36   case words line of
   37     [op, lab]
   38       | op `elem`
   39           [ "goto", "ifeq", "ifne", "iflt", "ifle", "ifgt", "ifge"
   40           , "if_icmpeq", "if_icmpne", "if_icmplt", "if_icmple"
   41           , "if_icmpgt", "if_icmpge"
   42           ] ->
   43           case M.lookup lab table of
   44             Just n  -> op ++ " " ++ show n
   45             Nothing -> error ("Unknown label: " ++ lab)
   46     _ -> line
   47 
   48 resolveLabelTable :: [String] -> (LabelTable, [String])
   49 resolveLabelTable = go 0 M.empty []
   50   where
   51     go _ table acc [] = (table, reverse acc)
   52     go pc table acc (line:rest)
   53       | isDirective line =
   54           go pc table (line:acc) rest
   55       | otherwise =
   56           case parseLabel line of
   57             Just lab -> go pc (M.insert lab pc table) acc rest
   58             Nothing  -> go (pc + 1) table (line:acc) rest
   59 
   60 resolveLabels :: CompilerData -> CompilerData
   61 resolveLabels (cp, defs, bytecode, symtab) =
   62   let (table, bcNoLab) = resolveLabelTable bytecode
   63       bcFinal          = map (correspondLine table) bcNoLab
   64   in (cp, defs, bcFinal, symtab)
   65 
   66 generateLabel :: CompilerData -> (String, CompilerData)
   67 generateLabel (cp, (heap, funcs, classes, enums, typedefs, counter), bc, sym) =
   68   let label = "L" ++ show counter
   69       newDefines = (heap, funcs, classes, enums, typedefs, counter + 1)
   70   in (label, (cp, newDefines, bc, sym))
