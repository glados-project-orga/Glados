    8 module Match (compileMatch) where

   10 import CompilerTypes (CompilerData)
   11 import CompilerTools (appendBody)
   12 import Ast (MatchStmt(..), MatchCase(..), Pattern(..), Literal(..), Expr(..))
   13 import Expr (compileExpr)
   14 import Labels (generateLabel)
   15 import EitherUtils (bindE)

   17 compileMatch :: MatchStmt -> CompilerData -> Either String CompilerData
   18 compileMatch (MatchStmt scrutinee cases) prog =
   19   bindE (compileExpr scrutinee prog) (\p1 ->
   20     let (lEnd, p2) = generateLabel p1
   21         (nonDef, mDef) = splitDefault cases
   22     in bindE (compileNonDefault lEnd nonDef p2) (\p3 ->
   23          bindE (compileDefaultIfAny lEnd mDef p3) (\p4 ->
   24            Right (emit (lEnd ++ ":") p4))))
   25 
   26 compileDefaultIfAny :: String -> Maybe MatchCase -> CompilerData -> Either String CompilerData
   27 compileDefaultIfAny _ Nothing p =
   28   Right (emit "pop" p)
   29 compileDefaultIfAny lEnd (Just c) p =
   30   compileDefault lEnd c p
   31 
   32 compileNonDefault :: String -> [MatchCase] -> CompilerData -> Either String CompilerData
   33 compileNonDefault _ [] p = Right p
   34 compileNonDefault lEnd (c:cs) p =
   35   bindE (compileOneCase lEnd c p) (\p' ->
   36     compileNonDefault lEnd cs p')
   37 
   38 compileOneCase :: String -> MatchCase -> CompilerData -> Either String CompilerData
   39 compileOneCase lEnd (MatchCase pat body) p =
   40   case pat of
   41     DefaultPattern -> Left "match: DefaultPattern must be last"
   42     LiteralPattern lit ->
   43       bindE (literalToComparableInt lit) (\litExpr ->
   44         let (lCase, p1) = generateLabel p
   45             (lNext, p2) = generateLabel p1
   46             p3 = emit "dup" p2
   47         in bindE (compileExpr litExpr p3) (\p4 ->
   48              let p5 = emit "isub" p4
   49                  p6 = emit ("ifeq " ++ lCase) p5
   50                  p7 = emit ("goto " ++ lNext) p6
   51                  p8 = emit (lCase ++ ":") p7
   52                  p9 = emit "pop" p8
   53              in bindE (compileExpr body p9) (\p10 ->
   54                   let p11 = emit ("goto " ++ lEnd) p10
   55                       p12 = emit (lNext ++ ":") p11
   56                   in Right p12)))
   57 
   58 compileDefault :: String -> MatchCase -> CompilerData -> Either String CompilerData
   59 compileDefault lEnd (MatchCase _ body) p =
   60   let p1 = emit "pop" p
   61   in bindE (compileExpr body p1) (\p2 ->
   62        Right (emit ("goto " ++ lEnd) p2))
   63 
   64 splitDefault :: [MatchCase] -> ([MatchCase], Maybe MatchCase)
   65 splitDefault [] = ([], Nothing)
   66 splitDefault (c:cs) =
   67   case matchPattern c of
   68     DefaultPattern -> (cs, Just c)
   69     _ ->
   70       let (xs, d) = splitDefault cs
   71       in (c:xs, d)
   72 
   73 emit :: String -> CompilerData -> CompilerData
   74 emit instr prog = appendBody prog [instr]
   75 
   76 literalToComparableInt :: Literal -> Either String Expr
   77 literalToComparableInt (IntLit n)  = Right (LitExpr (IntLit n))
   78 literalToComparableInt (BoolLit b) = Right (LitExpr (IntLit (if b then 1 else 0)))
   79 literalToComparableInt (CharLit c) = Right (LitExpr (IntLit (fromEnum c)))
   80 literalToComparableInt _ =
   81   Left "match: unsupported pattern literal type (only Int/Bool/Char supported)"
