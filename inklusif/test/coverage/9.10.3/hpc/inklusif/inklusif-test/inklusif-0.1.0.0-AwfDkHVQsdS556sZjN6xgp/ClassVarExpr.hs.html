    8 compileClassArrayVarExpr :: CompileExpr -> [String] -> Type -> Expr -> CompilerData -> Either String CompilerData
    9 compileClassArrayVarExpr re this t indexExprs prog = Right (appendBody prog (this))
   10     >>= \arrProg -> re indexExprs arrProg >>= \indexProg ->
   11     Right (appendBody indexProg [typePrefixVal t ++ "aload" ])

   13 compileBasicClassAccess :: CompileExpr -> [String] -> (String, ClassAccess) -> Type -> CompilerData -> Either String CompilerData
   14 compileBasicClassAccess re this (_, (ClassMethodCall (CallExpr cname args))) _ prog =
   15     compileMethExpr re this (CallExpr cname args) prog
   16 compileBasicClassAccess re this (_, (ClassArrayAccess arrName idxExpr)) st prog =
   17         compileClassArrayVarExpr re (this ++ ["getfield " ++ arrName]) st idxExpr prog
   18 compileBasicClassAccess _ this (_, (ClassVarAccess fldName)) _ prog =
   19     Right (appendBody prog (this ++ ["getfield " ++ fldName]))
   20 compileBasicClassAccess re this (_, (ClassClassAccess nclName cacc)) st prog =
   21     compileBasicClassAccess re (this ++["getfield " ++ nclName]) (nclName, cacc) st prog

   23 compileClassVarExpr :: CompileExpr -> (String, ClassAccess) -> CompilerData -> Either String CompilerData
   24 compileClassVarExpr re clVar@(cname, cacc) prog = getVarIndex cname prog >>=
   25         \idx -> convertToType (ClassVarExpr cname cacc) prog >>= \subType ->
   26             compileBasicClassAccess re ["aload " ++ show idx] clVar subType prog
