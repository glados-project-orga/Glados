   11 closeFunction :: CompilerData -> FunctionDecl -> Either String CompilerData
   12 closeFunction prog def = Right (appendDefines (appendBody prog ["}"]) [Function def] )
   13 
   14 addParams :: [Parameter] -> CompilerData -> Either String CompilerData
   15 addParams [] prog = Right prog
   16 addParams (Parameter name typ _:params) prog =
   17     addGoodTypeStore typ prog
   18     >>= \progWithParam -> Right (storeInSymbolTable name typ progWithParam)
   19     >>= \progWithStore -> addParams params progWithStore

   22 compileFunBody :: [Statement] -> Type -> CompilerData -> Either String CompilerData
   23 compileFunBody statement retype prog =  manageBody statement retype prog

   25 compileFunction :: FunctionDecl -> CompilerData->  Either String CompilerData
   26 compileFunction def@(FunctionDecl pos name params retype statement) prog@(_, defs,_ , _)
   27     | searchFunctions name defs /= Nothing =
   28         Left ((errPos pos) ++ "Function " ++ name ++ " is already defined.")
   29     | otherwise = Right (appendBody prog ["fun" ++ " " ++ name ++ " {"]) >>=
   30             \progWithOpenFun -> addParams params progWithOpenFun >>=
   31             \added_params_prog -> compileFunBody statement retype added_params_prog >>=
   32             \uncomplete_prog -> closeFunction uncomplete_prog def
