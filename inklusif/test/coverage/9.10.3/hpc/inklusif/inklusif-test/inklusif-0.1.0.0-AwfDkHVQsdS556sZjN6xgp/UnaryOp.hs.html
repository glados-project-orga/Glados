    8 module UnaryOp (compileUnaryOp) where
    9 
   10 import Ast (Expr(..), UnaryOp(..), Type(..))
   11 import CompilerTypes (CompilerData, CompileExpr)
   12 import CompilerTools (appendBody, convertToType, typePrefixVal)
   13 import SymbolTableUtils (getVarIndex, getVarType)
   14 
   15 compileUnaryOp :: CompileExpr -> Expr -> CompilerData -> Either String CompilerData
   16 compileUnaryOp compile (UnaryOpExpr op expr) prog =
   17     case op of
   18         Neg   -> compileNeg compile expr prog
   19         Not   -> compileNot compile expr prog
   20         PreInc -> compilePreIncDec compile expr 1 prog
   21         PreDec -> compilePreIncDec compile expr (-1) prog
   22         _ -> Left "Unary op: Not implemented, ref or deref"
   23 compileUnaryOp _ _ _ = Left "compileUnaryOp called with non-UnaryOp expression"
   24 
   25 compileNeg :: CompileExpr -> Expr -> CompilerData -> Either String CompilerData
   26 compileNeg compile expr prog =
   27     compile expr prog >>= \exprProg ->
   28     convertToType expr prog >>= \exprType ->
   29     emitNeg exprType exprProg
   30 
   31 emitNeg :: Type -> CompilerData -> Either String CompilerData
   32 emitNeg IntType prog    = Right $ appendBody prog ["ineg"]
   33 emitNeg LongType prog   = Right $ appendBody prog ["lneg"]
   34 emitNeg FloatType prog  = Right $ appendBody prog ["fneg"]
   35 emitNeg DoubleType prog = Right $ appendBody prog ["dneg"]
   36 emitNeg t _ = Left $ "Negation not supported for type: " ++ show t
   37 
   38 compileNot :: CompileExpr -> Expr -> CompilerData -> Either String CompilerData
   39 compileNot compile expr prog =
   40     compile expr prog >>= \exprProg ->
   41     convertToType expr prog >>= \exprType ->
   42     emitNot exprType exprProg
   43 
   44 emitNot :: Type -> CompilerData -> Either String CompilerData
   45 emitNot BoolType prog = Right $ appendBody prog ["iconst 1", "ixor"]
   46 emitNot IntType prog  = Right $ appendBody prog ["iconst 1", "ixor"]
   47 emitNot t _ = Left $ "not not supported for type: " ++ show t
   48 
   49 compilePreIncDec :: CompileExpr -> Expr -> Int -> CompilerData -> Either String CompilerData
   50 compilePreIncDec _ (VarExpr varName) sign prog =
   51     getVarIndex varName prog >>= \idx ->
   52     getVarType varName prog >>= \varType ->
   53     let prefix = typePrefixVal varType
   54         loadInstr = prefix ++ "load " ++ show idx
   55         storeInstr = prefix ++ "store " ++ show idx
   56         constInstr = prefix ++ "const " ++ show (abs sign)
   57         opInstr = if sign > 0
   58                   then prefix ++ "add"
   59                   else prefix ++ "sub"
   60     in Right $ appendBody prog [loadInstr, constInstr, opInstr, "dup", storeInstr]
   61 compilePreIncDec _ _ _ _ = Left "Pre-increment/decrement needs a variable"
