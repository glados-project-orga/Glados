   12 initialData :: CompilerData
   13 initialData = ([], (0, [], [], [], [], 0), [], [])
   14 
   15 compileDeclarations :: Declaration -> CompilerData -> Either String CompilerData
   16 compileDeclarations (Function fun) (cp, def, bc, _) = compileFunction fun (cp, def, bc, [])
   17 compileDeclarations (Class struct) prog = compileClass struct prog
   18 compileDeclarations (Enum enum) prog = compileEnum enum prog
   19 compileDeclarations (Typedef typedef) prog = compileTypedef typedef prog
   20 
   21 compilerMain :: Ast -> Either String CompilerData -> Either String CompilerData
   22 compilerMain _ (Left err) = Left err
   23 compilerMain [] (Right prog) = Right prog
   24 compilerMain (declaration:ast) (Right prog) =
   25   compilerMain ast (compileDeclarations declaration prog)
   26 
   27 compileAst :: Ast -> Either String (IO ())
   28 compileAst ast =
   29   compilerMain ast (Right initialData) >>= \prog ->
   30     let finalProg = resolveLabels prog
   31         content   = writeBytecode finalProg
   32     in Right (writeFile "binary.ink" content)
