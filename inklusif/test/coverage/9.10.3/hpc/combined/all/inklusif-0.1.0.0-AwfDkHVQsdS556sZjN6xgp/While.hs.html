   10 emitBranch :: String -> Expr -> Expr -> String -> CompilerData -> Either String CompilerData
   11 emitBranch op left right lEnd prog =
   12     compileExpr left prog >>= \p1 ->
   13     compileExpr right p1 >>= \p2 ->
   14     Right $ appendBody p2 ["if_icmp" ++ op ++ " " ++ lEnd]

   17 compileCondAsBranch :: Expr -> String -> CompilerData -> Either String CompilerData
   18 compileCondAsBranch (BinOpExpr op left right) lEnd prog =
   19     case op of
   20         LessEqual     -> emitBranch "gt" left right lEnd prog
   21         LessThan      -> emitBranch "ge" left right lEnd prog
   22         GreaterEqual  -> emitBranch "lt" left right lEnd prog
   23         GreaterThan   -> emitBranch "le" left right lEnd prog
   24         Equal         -> emitBranch "ne" left right lEnd prog
   25         NotEqual      -> emitBranch "eq" left right lEnd prog
   26         _             -> Left "Case not support in while"
   27 
   28 compileCondAsBranch _ _ _ = Left "Invalid conditions in while"
