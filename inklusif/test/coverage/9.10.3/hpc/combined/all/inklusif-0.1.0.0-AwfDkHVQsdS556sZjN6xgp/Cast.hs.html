    8 module Cast (compileCast) where
    9 
   10 import Ast (Expr(..), Type(..))
   11 import CompilerTypes (CompilerData, CompileExpr)
   12 import CompilerTools (appendBody, convertToType)
   13 
   14 compileCast :: CompileExpr -> Type -> Expr -> CompilerData -> Either String CompilerData
   15 compileCast compile targetType expr prog =
   16     compile expr prog >>= \exprProg ->
   17     convertToType expr prog >>= \sourceType ->
   18     emitConversion sourceType targetType exprProg
   19 
   20 emitConversion :: Type -> Type -> CompilerData -> Either String CompilerData
   21 
   22 emitConversion src tgt prog | src == tgt = Right prog
   23 
   24 emitConversion IntType CharType prog   = Right $ appendBody prog ["i2c"]
   25 emitConversion IntType LongType prog   = Right $ appendBody prog ["i2l"]
   26 emitConversion IntType FloatType prog  = Right $ appendBody prog ["i2f"]
   27 emitConversion IntType DoubleType prog = Right $ appendBody prog ["i2d"]
   28 
   29 emitConversion LongType IntType prog    = Right $ appendBody prog ["l2i"]
   30 emitConversion LongType FloatType prog  = Right $ appendBody prog ["l2f"]
   31 emitConversion LongType DoubleType prog = Right $ appendBody prog ["l2d"]
   32 
   33 emitConversion FloatType IntType prog    = Right $ appendBody prog ["f2i"]
   34 emitConversion FloatType LongType prog   = Right $ appendBody prog ["f2l"]
   35 emitConversion FloatType DoubleType prog = Right $ appendBody prog ["f2d"]
   36 
   37 emitConversion DoubleType IntType prog   = Right $ appendBody prog ["d2i"]
   38 emitConversion DoubleType LongType prog  = Right $ appendBody prog ["d2l"]
   39 emitConversion DoubleType FloatType prog = Right $ appendBody prog ["d2f"]
   40 
   41 emitConversion CharType IntType prog = Right $ appendBody prog ["c2i"]
   42 
   43 emitConversion BoolType IntType prog = Right prog
   44 
   45 emitConversion _ _ _ = Left $ "Unsupported cast"
