   10 import Ast (Literal(..))
   11 import CompilerTypes (CompilerData)
   12 import CompilerTools (appendBody)
   13 
   14 compileLiteralExpr :: Literal -> CompilerData -> Either String CompilerData
   15 compileLiteralExpr (IntLit n) prog = Right $ appendBody prog ["iconst " ++ show n]
   16 compileLiteralExpr (LongLit n) prog = Right $ appendBody prog ["lconst " ++ show n]
   17 compileLiteralExpr (FloatLit x) prog = Right $ appendBody prog ["fconst " ++ show x]
   18 compileLiteralExpr (DoubleLit x) prog = Right $ appendBody prog ["dconst " ++ show x]
   19 compileLiteralExpr (BoolLit True) prog = Right $ appendBody prog ["iconst 1"]
   20 compileLiteralExpr (BoolLit False) prog = Right $ appendBody prog ["iconst 0"]
   21 compileLiteralExpr (CharLit c) prog = Right $ appendBody prog ["cconst " ++ "'" ++ [c] ++ "'"]
   22 compileLiteralExpr (StringLit s) prog = Right $ appendBody prog ["sconst \"" ++ s ++"\""]
