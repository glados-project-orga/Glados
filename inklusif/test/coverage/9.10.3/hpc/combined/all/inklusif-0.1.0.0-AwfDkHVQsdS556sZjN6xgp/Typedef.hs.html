    1 module Typedef (compileTypedef, getTypedefType, typeNameExists) where
    2 import Ast (TypedefDecl(..), Declaration(..), Type(..))
    3 import CompilerTypes (CompilerData)
    4 import CompilerError (errPos)
    5 import CompilerTools (appendDefines)
    6 import Data.List (find)

    8 typeNameExists :: String -> [TypedefDecl] -> Bool
    9 typeNameExists searched typedefs =
   10     any (\(TypedefDecl _ _ name) -> name == searched) typedefs
   11 
   12 getTypedefType :: String -> [TypedefDecl] -> Maybe Type
   13 getTypedefType searched typedefs =
   14     typedefOriginal <$> find (\td -> typedefAlias td == searched) typedefs
   15 
   16 compileTypedef :: TypedefDecl  -> CompilerData -> Either String CompilerData
   17 compileTypedef typedef@(TypedefDecl pos _ name) prog@(_, (_, _, _, _, typedefs, _), _, _)
   18     | typeNameExists name typedefs =
   19         Left ((errPos pos) ++ "Typedef " ++ name ++ " is already defined.")
   20     | otherwise = Right (appendDefines prog [(Typedef typedef)])
