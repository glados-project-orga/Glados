    8 enumNameExists :: String -> [EnumDecl] -> Bool
    9 enumNameExists searched enums =
   10     any (\enum -> enumName enum == searched) enums
   11 
   12 enumFieldExists :: [EnumField] -> [EnumDecl] -> Bool
   13 enumFieldExists fields enums =
   14     (any (\field -> getEnumValue enums (declName field) /= Nothing) fields)
   15 
   16 fillEnumValues :: [EnumField] -> Int -> [EnumField]
   17 fillEnumValues fields startVal =
   18     zipWith (\field val -> field {declValue = Just val}) fields [startVal..]

   20 compileEnum :: EnumDecl -> CompilerData -> Either String CompilerData
   21 compileEnum (EnumDecl pos name []) _ =
   22     Left ((errPos pos) ++ "Enum " ++ name ++ " must have at least one field.")
   23 compileEnum enum@(EnumDecl pos name fields@(field:_)) prog@(_, (_, _, _, enums, _, _), _, _)
   24     | enumNameExists name enums =
   25         Left ((errPos pos) ++ "Enum " ++ name ++ " is already defined.")
   26     | enumFieldExists fields enums =
   27         Left ((errPos pos) ++ "Enum " ++ name ++ " has fields that are already defined.")
   28     | otherwise = Right $ appendDefines prog [(Enum enum {enumDecl = filledFields})]
   29         where filledFields = fillEnumValues fields startVal
   30               startVal = fromMaybe 0 (declValue field)
