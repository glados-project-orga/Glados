    9 validExprForTask :: SearchTypes -> SearchTypes -> CompilerData -> Either String CompilerData
   10 validExprForTask val sexpr@(SearchExpr expr) prog |
   11     validAssignmentType val sexpr prog = compileExpr expr prog
   12     | otherwise = Left ("Invalid expression type for array index."
   13         ++ show (convertToType expr prog) ++ "." ++ " Expected " ++ show val ++ ".")
   14 validExprForTask  _ _ _ = Left "Invalid search type for array index."

   16 compileArrayIndex :: ArrayIndexExpr -> CompilerData -> Either String CompilerData
   17 compileArrayIndex  (ArrayIndexExpr name index val) prog =
   18     getVarIndex name prog >>= \varIndex ->
   19     Right (appendBody prog ["aload " ++ show varIndex]) >>= \loadProg ->
   20     validExprForTask (srch IntType) (srch index) loadProg >>= \indexProg ->
   21     getVarType name indexProg >>= \arrayVal -> getArraySubType arrayVal >>= \subType ->
   22     validExprForTask (srch subType) (srch val) indexProg >>=
   23     \valProg -> Right $ appendBody valProg [(getTypePrefix (show subType)) ++ "astore"]

   25 basicAssignment  :: Assignment -> CompilerData -> Either String CompilerData
   26 basicAssignment (Assignment target@(VarExpr name) value) prog
   27     | validAssignmentType (srch target) (srch value) prog =
   28         compileExpr (value) prog >>= \value_prog -> getVarIndex name value_prog >>=
   29         \varIndex -> getVarType name value_prog >>= \varVal -> Right (appendBody value_prog 
   30         [typePrefixVal varVal ++ "store " ++ show varIndex])
   31     | otherwise = Left ("Invalid assignment types for " ++ name ++ ": " ++ (show (convertToType value prog)) ++ ".")
   32 basicAssignment target _ = Left ("Unknown assignment target. " ++ show target)

   35 multipleFieldAccess :: ClassAccess -> CompilerData -> Either String (String, CompilerData)
   36 multipleFieldAccess (ClassVarAccess varName) prog = Right (varName, prog)
   37 multipleFieldAccess (ClassClassAccess nclName cacc) prog = Right (appendBody prog ["getfield " ++ nclName]) >>=
   38     \n_prog -> multipleFieldAccess cacc n_prog
   39 multipleFieldAccess _ _ = Left "Unknown assignment target."

   41 maybeCreateArray :: Expr -> CompilerData -> Either String CompilerData
   42 maybeCreateArray expr prog = case convertToType expr prog of
   43     Right (ArrayType (ArrayVar _ (LitExpr (IntLit len)))) -> Right (appendBody prog ["iconst " ++ show len, "newarray"])
   44     Right _ -> Right prog
   45     Left err -> Left err

   47 classAssignment :: Expr -> Expr -> CompilerData -> Either String CompilerData
   48 classAssignment target@(ClassVarExpr objName access) value prog |
   49     validAssignmentType (srch target) (srch value) prog =
   50         getVarIndex objName prog >>=
   51         \objIndex -> (Right (appendBody prog ["aload " ++ show objIndex])) >>=
   52         \objProg -> multipleFieldAccess access objProg >>=
   53         \(name, fieldProg) -> maybeCreateArray target fieldProg >>= \arrayProg -> compileExpr value arrayProg >>=
   54         \valueProg -> Right $ appendBody valueProg ["putfield " ++ name]
   55     | otherwise = errName >>= \ern -> Left ("Invalid assignment types for field '" ++ (fst ern) ++ "': " ++
   56         "cannot assign " ++ (show (convertToType value prog)) ++ " to field of type "
   57         ++ show (convertToType target prog) ++ ". TypeEq result: " ++ show ((convertToType value prog) `typeEq` (convertToType target prog))) 
   58         where errName = (multipleFieldAccess access prog)
   59 classAssignment _ _ _ = Left "Unknown assignment target."

   61 compileAssignment :: Assignment -> CompilerData -> Either String CompilerData
   62 compileAssignment (Assignment target@(VarExpr _) value) prog = basicAssignment (Assignment target value) prog
   63 compileAssignment (Assignment (ArrayVarExpr arrName indexExpr) value) prog =
   64     compileArrayIndex (ArrayIndexExpr arrName indexExpr value) prog
   65 compileAssignment (Assignment target@(ClassVarExpr _ _) value) prog =
   66     classAssignment target value prog
   67 compileAssignment _ _ = Left "Unknown assignment target."
