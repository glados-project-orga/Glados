   15 pushArg :: CompileExpr -> Expr -> Parameter -> CompilerData -> Either String CompilerData
   16 pushArg re expr (Parameter _ parType _) prog
   17     | validAssignmentType (srch parType) (srch expr) prog = re expr prog
   18     | otherwise = Left "Invalid parameter type in function call."
   19 
   20 pushArgs :: CompileExpr -> [Expr] -> [Parameter] -> CompilerData -> Either String CompilerData
   21 pushArgs _ (_:_) [] _ = Left "Too much arguments in function call."
   22 pushArgs _ [] (_:_) _ = Left "Not enough arguments in function call."
   23 pushArgs _ [] [] prog = Right prog
   24 pushArgs re (e:es) (p:ps) prog = pushArgs re es ps prog >>= pushArg re e p

   28 compileMethExpr:: CompileExpr -> [String] -> CallExpr -> CompilerData -> Either String CompilerData
   29 compileMethExpr re this (CallExpr name exprs) prog@(_, defs, _, _) = eitherParams >>=
   30     \params -> pushArgs re exprs params prog >>= \incomplete_prog -> (Right (appendBody incomplete_prog this)) >>=
   31     \n_prog -> Right (appendBody n_prog ["invokestatic " ++ name])
   32     where eitherParams = case searchFunctions name defs of
   33             Just (FunctionDecl _ _ foundParams _ _) -> case foundParams of
   34                 _:thisSkip -> Right thisSkip
   35                 _ -> Left ("Method " ++ name ++ " missing 'this' parameter.")
   36             Nothing -> Left ("Function " ++ name ++ " not found.")
   37 
   38 compileCallExpr:: CompileExpr -> CallExpr -> CompilerData -> Either String CompilerData
   39 compileCallExpr compileExpr (CallExpr "write" args) prog
   40     | length args /= 2 = Left "Error exepected two arguments for write"
   41     | otherwise =
   42         case args of
   43             (x:(LitExpr (IntLit n)): _) ->
   44                 if n <= 0 then Left "exepected 1 or 2 for output"
   45                 else
   46                     compileExpr x prog >>= \p1 ->
   47                     Right (appendBody p1 ["invoke_write " ++ show n])
   48             _ -> Left "Error: write expects (expr, int)"
   49 
   50 compileCallExpr _ (CallExpr "open" args) prog
   51     | length args /= 1 =  Left "Error exepected one argument for open"
   52     | otherwise =
   53         case args of
   54             [LitExpr (StringLit file)] -> 
   55                 let prog1 = appendBody prog ["sconst \"" ++ file ++ "\""] 
   56                     prog2 = appendBody prog1 ["invoke_open"]
   57                 in Right prog2
   58             _ -> Left "Error: open expects a string literal"
   59 
   60 compileCallExpr re (CallExpr name exprs) prog@(_, defs, _, _) = eitherParams >>=
   61     \params -> pushArgs re exprs params prog >>=
   62     \n_prog -> Right (appendBody n_prog ["invokestatic " ++ name])
   63     where eitherParams = case searchFunctions name defs of
   64             Just (FunctionDecl _ _ foundParams _ _) -> Right (foundParams)
   65             Nothing -> Left ("Function " ++ name ++ " not found.")
