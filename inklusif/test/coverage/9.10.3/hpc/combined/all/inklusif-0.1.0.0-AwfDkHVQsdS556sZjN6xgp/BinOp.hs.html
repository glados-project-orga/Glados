    8 module BinOp (compileBinOpExpr) where
    9 
   10 import Labels (generateLabel)
   11 import Ast (Expr(..), BinOp(..), Type(..))
   12 import CompilerTypes (CompilerData, CompileExpr)
   13 import CompilerTools (appendBody, convertToType, typePrefixVal)
   14 import SymbolTableUtils (getVarIndex, getVarType)
   15 
   16 compileBinOpExpr :: CompileExpr -> Expr -> CompilerData -> Either String CompilerData
   17 compileBinOpExpr compile (BinOpExpr op left right) prog =
   18     case op of
   19         AddEqual -> compileCompoundAssign compile Add left right prog
   20         SubEqual -> compileCompoundAssign compile Sub left right prog
   21         MulEqual -> compileCompoundAssign compile Mul left right prog
   22         DivEqual -> compileCompoundAssign compile Div left right prog
   23         ModEqual -> compileCompoundAssign compile Mod left right prog
   24         _ -> compile left prog >>= \progLeft ->
   25              compile right progLeft >>= \progRight ->
   26              convertToType left prog >>= \exprType ->
   27              emitBinOp op exprType progRight
   28 compileBinOpExpr _ _ _ = Left "compileBinOpExpr called with non-BinOp expression"
   29 
   30 compileCompoundAssign :: CompileExpr -> BinOp -> Expr -> Expr -> CompilerData -> Either String CompilerData
   31 compileCompoundAssign compile op (VarExpr varName) right prog =
   32     getVarIndex varName prog >>= \idx ->
   33     getVarType varName prog >>= \varType ->
   34     let loadInstr = typePrefixVal varType ++ "load " ++ show idx
   35         storeInstr = typePrefixVal varType ++ "store " ++ show idx
   36     in Right (appendBody prog [loadInstr]) >>= \progLoad ->
   37        compile right progLoad >>= \progRight ->
   38        emitBinOp op varType progRight >>= \progOp ->
   39        Right (appendBody progOp [storeInstr])
   40 compileCompoundAssign _ _ _ _ _ =
   41     Left "Compound assignment requires a variable on the left side"
   42 
   43 emitBinOp :: BinOp -> Type -> CompilerData -> Either String CompilerData
   44 
   45 emitBinOp Add t prog = Right $ appendBody prog [typePrefixVal t ++ "add"]
   46 emitBinOp Sub t prog = Right $ appendBody prog [typePrefixVal t ++ "sub"]
   47 emitBinOp Mul t prog = Right $ appendBody prog [typePrefixVal t ++ "mul"]
   48 emitBinOp Div t prog = Right $ appendBody prog [typePrefixVal t ++ "div"]
   49 emitBinOp Mod t prog = Right $ appendBody prog [typePrefixVal t ++ "rem"]
   50 
   51 emitBinOp Equal t prog = Right $ emitComparison t "eq" prog
   52 emitBinOp NotEqual t prog = Right $ emitComparison t "ne" prog
   53 emitBinOp LessThan t prog = Right $ emitComparison t "lt" prog
   54 emitBinOp GreaterThan t prog = Right $ emitComparison t "gt" prog
   55 emitBinOp LessEqual t prog = Right $ emitComparison t "le" prog
   56 emitBinOp GreaterEqual t prog = Right $ emitComparison t "ge" prog
   57 
   58 emitBinOp And _ prog = Right $ appendBody prog ["iand"]
   59 emitBinOp Or _ prog = Right $ appendBody prog ["ior"]
   60 
   61 emitBinOp op _ _ = Left $ "BinOp not implemented: " ++ show op
   62 
   63 emitComparison :: Type -> String -> CompilerData -> CompilerData
   64 emitComparison (IntType) cond prog = emitIntComparison (IntType) cond prog
   65 emitComparison t cond prog = emitCmpThenBranch t cond prog
   66 
   67 emitIntComparison :: Type -> String -> CompilerData -> CompilerData
   68 emitIntComparison _ cond prog =
   69     let (lTrue,  prog1) = generateLabel prog
   70         (lEnd,   prog2) = generateLabel prog1
   71         prog3 = appendBody prog2 ["if_icmp" ++ cond ++ " " ++ lTrue]
   72         prog4 = appendBody prog3 ["iconst 0"]
   73         prog5 = appendBody prog4 ["goto " ++ lEnd]
   74         prog6 = appendBody prog5 [lTrue ++ ":"]
   75         prog7 = appendBody prog6 ["iconst 1"]
   76         prog8 = appendBody prog7 [lEnd ++ ":"]
   77     in prog8
   78 
   79 
   80 emitCmpThenBranch :: Type -> String -> CompilerData -> CompilerData
   81 emitCmpThenBranch t cond prog =
   82     let (lTrue,  prog1) = generateLabel prog
   83         (lEnd,   prog2) = generateLabel prog1
   84         prog3 = appendBody prog2 [cmpInstr t]
   85         prog4 = appendBody prog3 ["if" ++ cond ++ " " ++ lTrue]
   86         prog5 = appendBody prog4 ["iconst 0"]
   87         prog6 = appendBody prog5 ["goto " ++ lEnd]
   88         prog7 = appendBody prog6 [lTrue ++ ":"]
   89         prog8 = appendBody prog7 ["iconst 1"]
   90         prog9 = appendBody prog8 [lEnd ++ ":"]
   91     in prog9
   92 
   93 
   94 cmpInstr :: Type -> String
   95 cmpInstr (LongType) = "lcmp"
   96 cmpInstr (FloatType) = "fcmpl"
   97 cmpInstr (DoubleType) = "dcmpl"
   98 cmpInstr _ = "lcmp"
