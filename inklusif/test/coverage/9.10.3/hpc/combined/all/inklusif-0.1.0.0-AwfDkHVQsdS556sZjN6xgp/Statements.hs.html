   19 compileStatement :: Statement -> CompilerData -> Either String CompilerData
   20 compileStatement (VarDeclStmt vr_dcl) layer = compileVarDecl vr_dcl layer
   21 compileStatement (AssignmentStmt assign) layer = compileAssignment assign layer
   22 compileStatement (IfStatement if_st) layer = compileIf compileStatement if_st layer
   23 compileStatement (WhileStatement while) layer = compileWhile compileStatement while layer
   24 compileStatement (ForStatement for) layer = compileFor compileStatement for layer
   25 compileStatement (ForEachStatement for_each) layer = compileForEach compileStatement for_each layer
   26 compileStatement (MatchStatement match) layer = compileMatch match layer
   27 compileStatement (TryCatchStatement tryCatch) layer = compileTryCatch tryCatch layer
   28 compileStatement (ThrowStatement throw) layer = compileThrow throw layer
   29 compileStatement (ExprStatement (ExprStmt expr)) layer = compileExpr expr layer
   30 compileStatement (ReturnStatement ret) layer = compileReturn ret layer
   31 compileStatement _ _ = Left "Unsupported statement type in compileStatement"
   32 
   33 checkReturnType :: [Statement] -> Type -> CompilerData -> Either String CompilerData
   34 checkReturnType [] VoidType prog = Right (appendBody prog ["return"])
   35 checkReturnType [] _ _ = Left "Function missing return statement in non-void function."
   36 checkReturnType ((ReturnStatement (ReturnStmt expr)):_) retype prog
   37     | validAssignmentType (srch retype) (srch expr) prog = Right (prog)
   38     | otherwise = Left ("Invalid return Type " ++ (show (convertToType expr prog))  ++ " in function.")
   39 checkReturnType (_:stmts) retype prog = checkReturnType stmts retype prog

   41 compileStatements :: [Statement] ->  Either String CompilerData -> Either String CompilerData
   42 compileStatements _ (Left err) = Left err
   43 compileStatements stmts (Right layer) =
   44   compileBlock compileStatement stmts layer

   46 manageBody :: [Statement] -> Type -> CompilerData -> Either String CompilerData
   47 manageBody stmts t prog = compileStatements stmts (Right prog)
   48   >>= \newBodyProg -> checkReturnType stmts t newBodyProg
