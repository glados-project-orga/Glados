    8 module Var (compileVarExpr) where
    9 
   10 import CompilerTypes (CompilerData, SymInfo(..))
   11 import SymbolTableUtils (getVar)
   12 import CompilerTools (appendBody, typePrefixVal, getEnumValue)
   13 
   14 compileVarExpr :: String -> CompilerData -> Either String CompilerData
   15 compileVarExpr varName prog@(_, (_, _, _, enums, _, _), _, _)
   16     = case getEnumValue enums varName of
   17         Just enumVal -> Right (appendBody prog ["iconst " ++ show enumVal])
   18         _ -> getVar varName prog >>= \varVal -> case varVal of
   19                 (SymInfo _ _ (Just cindex) _) -> Right (appendBody prog ["ldc " ++ show cindex])
   20                 (SymInfo index t _ _)-> Right (appendBody prog [typePrefixVal t ++ "load " ++ show index])
