    8 module If (compileIf) where
    9 
   10 import CompilerTypes (CompilerData)
   11 import CompilerTools (appendBody)
   12 import Ast (IfStmt(..), Statement)
   13 import Expr (compileExpr)
   14 import Labels (generateLabel)
   15 import CompileBlock (compileBlock)
   16 
   17 compileIf
   18   :: (Statement -> CompilerData -> Either String CompilerData)
   19   -> IfStmt
   20   -> CompilerData
   21   -> Either String CompilerData
   22 compileIf compileStmt ifs prog =
   23   case compileExpr (ifCondition ifs) prog of
   24     Left err -> Left err
   25     Right p1 ->
   26       case ifElseBody ifs of
   27         Nothing      -> compileIfNoElse compileStmt ifs p1
   28         Just elseBlk -> compileIfWithElse compileStmt ifs elseBlk p1
   29 
   30 compileIfNoElse
   31   :: (Statement -> CompilerData -> Either String CompilerData)
   32   -> IfStmt
   33   -> CompilerData
   34   -> Either String CompilerData
   35 compileIfNoElse compileStmt ifs prog =
   36   let (lEnd, p1) = generateLabel prog
   37       p2 = emit ("ifeq " ++ lEnd) p1
   38   in case compileBlock compileStmt (ifThenBody ifs) p2 of
   39        Left err -> Left err
   40        Right p3 -> Right (emit (lEnd ++ ":") p3)
   41 
   42 compileIfWithElse
   43   :: (Statement -> CompilerData -> Either String CompilerData)
   44   -> IfStmt
   45   -> [Statement]
   46   -> CompilerData
   47   -> Either String CompilerData
   48 compileIfWithElse compileStmt ifs elseBlk prog =
   49   let (lElse, p1) = generateLabel prog
   50       (lEnd,  p2) = generateLabel p1
   51       p3 = emit ("ifeq " ++ lElse) p2
   52   in case compileBlock compileStmt (ifThenBody ifs) p3 of
   53        Left err -> Left err
   54        Right p4 ->
   55          let p5 = emit ("goto " ++ lEnd) p4
   56              p6 = emit (lElse ++ ":") p5
   57          in case compileBlock compileStmt elseBlk p6 of
   58               Left err -> Left err
   59               Right p7 -> Right (emit (lEnd ++ ":") p7)
   60 
   61 emit :: String -> CompilerData -> CompilerData
   62 emit instr prog = appendBody prog [instr]
