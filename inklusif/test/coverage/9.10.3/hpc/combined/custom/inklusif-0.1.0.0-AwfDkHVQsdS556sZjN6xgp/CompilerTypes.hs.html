   97 instance TypeEq Type (Either String String) where
   98     typeEq _ (Left _)      = False
   99     typeEq val (Right str)     = typeEq val str
  100 
  101 instance TypeEq Type (Maybe Type) where
  102     typeEq _ (Nothing)      = False
  103     typeEq val (Just typ)     =  val == typ
  104 
  105 instance TypeEq Type (Either String Type) where
  106     typeEq _ (Left _)      = False
  107     typeEq val (Right cmplVal)     = val == cmplVal
  108 
  109 instance TypeEq (Either String Type) (Either String Type) where
  110     typeEq (Left _) _ = False
  111     typeEq _ (Left _) = False
  112     typeEq (Right val1) (Right val2) = typeEq val1 val2
  113 
  114 instance TypeEq TypeNormalized TypeNormalized where
  115     typeEq (TypeNorm t1) (TypeNorm t2) = t1 == t2
  116     typeEq (LitNorm l1) (LitNorm l2)   = l1 == l2
  117     typeEq _ _                         = False
  118 
  119 instance TypeEq Type Type where
  120     typeEq (ArrayType (ArrayVar CharType _)) StringType = True
  121     typeEq (ArrayType (ArrayVar t1 _)) (ArrayType (ArrayVar t2 _)) = typeEq t1 t2
  122     typeEq (CustomType name1) (CustomType name2)  = name1 == name2
  123     typeEq (LongType) (IntType) = True
  124     typeEq t1 t2                      = t1 == t2
  125 
  126 instance TypeEq Type String where
  127     typeEq IntType "int"          = True
  128     typeEq FloatType "float"      = True
  129     typeEq DoubleType "double"    = True
  130     typeEq BoolType "bool"        = True
  131     typeEq CharType "char"        = True
  132     typeEq (ArrayType (ArrayVar CharType _)) "string" = True
  133     typeEq (ArrayType (ArrayVar t _)) arrayVar = typeEq t arrayVar
  134     typeEq VoidType "void"        = True
  135     typeEq (CustomType name) str  = name == str
  136     typeEq _ _                    = False
  137 
  138 instance TypeEq Type Literal where
  139     typeEq (IntType) (IntLit _)          = True
  140     typeEq (DoubleType) (DoubleLit _)    = True
  141     typeEq (FloatType) (FloatLit _)      = True
  142     typeEq (BoolType) (BoolLit _)        = True
  143     typeEq (CharType) (CharLit _)        = True
  144     typeEq (ArrayType (ArrayVar CharType _)) (StringLit _) = True
  145     typeEq _ _                             = False
