  134 -- Les types je pense c'est pas dur à expliquer
  135 data Type
  136   = IntType
  137   | FloatType -- j'espère on le fait pas
  138   | StringType
  139   | DoubleType
  140   | CharType
  141   | LongType
  142   | BoolType
  143   | LambdaType LambdaVar
  144   | ArrayType ArrayVar
  145   | CustomType String  -- (struct, enum, typedef)
  146   | VoidType -- iel
  147   deriving (Eq)
  148 
  149 instance Show Type where
  150   show IntType = "int"
  151   show FloatType = "float"
  152   show StringType = "string"
  153   show CharType = "char"
  154   show BoolType = "bool"
  155   show LongType = "long"
  156   show (LambdaType _) = "lambda"
  157   show DoubleType = "double"
  158   show (ArrayType (ArrayVar t s)) = "array " ++ show t ++ " " ++ show s
  159   show (CustomType s) = s
  160   show VoidType = "void"

  295 data Expr
  296   = LitExpr Literal

  298   | ArrayVarExpr String Expr
  299   | ClassVarExpr String ClassAccess
  300   | BinOpExpr BinOp Expr Expr
  301   | UnaryOpExpr UnaryOp Expr
  302   | CallExpression CallExpr
  303   | ClassConstructorExpr String [Expr]
  304   | MethodCallExpression MethodCallExpr
  305   | AssignmentExpr Assignment
  306   | ArrayLiteral [Expr]
  307   | ArrayAssignement ArrayIndexExpr
  308   | CastExpr Type Expr
  309   | FieldAccessExpression FieldAccessExpr
  310   | StructLiteral [(String, Expr)]
  311   | Lambda [Parameter] [Statement]
  312   deriving (Show, Eq)
