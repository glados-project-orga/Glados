    8 module Expr (compileExpr) where
    9 
   10 import Ast (Expr(..), Type(..), ArrayVar(..))
   11 import CompilerTypes (CompilerData, CompileExpr)
   12 import CompilerTools (getLitArrayType)
   13 import CompileLiteral (compileLiteralExpr)
   14 import BinOp (compileBinOpExpr)
   15 import ArrayLiteral (compileArrayLiteral)
   16 import Call (compileCallExpr)
   17 import Var (compileVarExpr)
   18 import ArrayVarExpr (compileArrayVarExpr)
   19 import ClassVarExpr (compileClassVarExpr)
   20 import Cast (compileCast)
   21 import ClassConstructorExpr (compileClassConstructorExpr)
   22 import UnaryOp (compileUnaryOp)
   23 
   24 prepareArrayLiteral :: CompileExpr -> [Expr] -> CompilerData -> Either String CompilerData
   25 prepareArrayLiteral _ [] prog = Right prog
   26 prepareArrayLiteral re exprs prog = arrayType >>= \at -> compileArrayLiteral re exprs at prog
   27     where arrayType = getLitArrayType exprs prog >>= \t -> case t of
   28             (ArrayType (ArrayVar at _)) -> Right at
   29             _ -> Left "Expected array type"
   30 
   31 compileExpr :: Expr -> CompilerData -> Either String CompilerData
   32 compileExpr (LitExpr lit) prog = compileLiteralExpr lit prog
   33 compileExpr (ArrayLiteral exprs) prog = prepareArrayLiteral compileExpr exprs prog
   34 compileExpr (CallExpression call) prog = compileCallExpr compileExpr call prog
   35 compileExpr (ClassVarExpr cname cacc) prog = compileClassVarExpr compileExpr (cname, cacc) prog
   36 compileExpr (ClassConstructorExpr cname exprs) prog = compileClassConstructorExpr (cname, exprs) prog
   37 compileExpr (VarExpr expr) prog = compileVarExpr expr prog
   38 compileExpr (ArrayVarExpr nam idxe) prog = compileArrayVarExpr compileExpr (nam, idxe) prog
   39 compileExpr expr@(BinOpExpr _ _ _) prog = compileBinOpExpr compileExpr expr prog
   40 compileExpr (CastExpr t expr) prog = compileCast compileExpr t expr prog
   41 compileExpr expr@(UnaryOpExpr _ _) prog = compileUnaryOp compileExpr expr prog
   42 compileExpr _ _ = Left "Expression type not implemented yet"
