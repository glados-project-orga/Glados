    8 module CompileBlock (compileBlock) where
    9 
   10 import Ast (Statement)
   11 import CompilerTypes (CompilerData)
   12 
   13 compileBlock
   14   :: (Statement -> CompilerData -> Either String CompilerData)
   15   -> [Statement]
   16   -> CompilerData
   17   -> Either String CompilerData
   18 compileBlock _ [] prog = Right prog
   19 compileBlock compileStmt (s:ss) prog =
   20   case compileStmt s prog of
   21     Left err -> Left err
   22     Right prog' -> compileBlock compileStmt ss prog'
