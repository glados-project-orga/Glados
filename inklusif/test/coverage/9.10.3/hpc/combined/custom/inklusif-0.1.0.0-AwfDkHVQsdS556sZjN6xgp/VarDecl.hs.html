   17 storeInSymbolTable :: String -> Type -> CompilerData -> CompilerData
   18 storeInSymbolTable name t prog@(_, _, _, symTable) =
   19     appendSymbolTable prog [(name, (SymInfo (localindex) t Nothing False))]
   20         where localindex = length symTable
   21 
   22 addGoodTypeStore :: Type -> CompilerData -> Either String CompilerData
   23 addGoodTypeStore t prog@(_, _, _, symTable) =
   24     Right (appendBody prog [typePrefixVal t ++ "store " ++ show (localindex)])
   25         where localindex = length symTable

   27 storeVar :: String -> Type -> Expr -> CompilerData -> Either String CompilerData
   28 storeVar name t value prog =
   29     compileExpr value prog >>= \expryProg ->
   30     addGoodTypeStore t expryProg >>= \valBodyProg ->
   31     Right(storeInSymbolTable name t valBodyProg)
   32 
   33 storeConstVar :: String -> Type -> Expr -> CompilerData -> Either String CompilerData
   34 storeConstVar name t value prog@(header, _, _, symTable) =
   35     Right (appendSymbolTable prog [(name, (SymInfo (localindex) t (Just headerindex) False))])
   36     >>= \progWithVar ->
   37     addValToHeader progWithVar value
   38         where localindex = length symTable
   39               headerindex = length header
   40 
   41 storeArrayVar :: String -> Type -> ArrayVar -> Expr -> CompilerData -> Either String CompilerData
   42 storeArrayVar name t (ArrayVar at len) (ArrayLiteral exprs) prog =
   43              compileExpr len prog
   44             >>= \lenProg -> Right (appendBody lenProg ["newarray"])
   45             >>= \newArrayProg -> compileArrayLiteral compileExpr exprs at newArrayProg
   46             >>= \newArrayWithElementsProg -> addGoodTypeStore t newArrayWithElementsProg
   47             >>= \storedProg -> Right (storeInSymbolTable name t storedProg)
   48 storeArrayVar _ _ _ expr _ = Left ("Invalid array variable declaration with. " ++ show expr)

   50 getCustomType :: Type -> Defines -> Either String Type
   51 getCustomType ct@(CustomType cname) defs@(_, _, _, _, typedefs, _)
   52     | typeNameExists cname typedefs = Right $ (maybe VoidType id (getTypedefType cname typedefs))
   53     | isClassDefined cname defs = Right $ ct
   54     | otherwise = Left ("Type " ++ cname ++ " does not exist.")
   55 getCustomType t _ = Right t
   56 
   57 getTrueType :: Type -> Defines -> Type
   58 getTrueType t@(CustomType cname) (_, _, _, _, typedefs, _) =
   59     case getTypedefType cname typedefs of
   60         Just typedefType -> typedefType
   61         Nothing -> t
   62 getTrueType t _ = t

   64 isSameType :: Type -> Expr -> CompilerData -> Bool
   65 isSameType t expr prog@(_, defines, _, _) = case convertToType expr prog of
   66     Right convertedType -> trueType `typeEq` convertedType
   67         where trueType = getTrueType t defines
   68     Left _ -> False

   70 compileVarDecl :: VarDecl -> CompilerData -> Either String CompilerData
   71 compileVarDecl (VarDecl _ _ _ True True) _ =
   72     Left "Cannot have a variable that is both constant and reference."
   73 compileVarDecl (VarDecl name t value True _) prog = storeConstVar name t value prog
   74 compileVarDecl (VarDecl _ _ _ _ True) prog = Right prog
   75 compileVarDecl (VarDecl name t value _ _) prog@(_,  defines, _, _)
   76     | isRight (getVar name prog) = Left ("Variable " ++ name ++ " is already declared.")
   77     | isSameType t value prog = case t of
   78         ArrayType arr -> storeArrayVar name t arr value prog
   79         CustomType _ -> getCustomType t defines >>=
   80             \ct -> storeVar name ct value prog
   81         _ -> storeVar name t value prog
   82     | otherwise = Left ("Variable declaration type does not match assigned value type." ++ 
   83         case convertToType value prog of
   84             Right ct -> " Expected " ++ show t ++ ", got " ++ show ct ++ "."
   85             Left err -> " (Could not determine type: " ++ err ++ ")")
